What are the most popular routes where the start and end stations are NOT the same?

# Create a new data frame that includes only non-round-trip trips
non_round_trips <- bike_data %>% filter(Start.Station != End.Station)

# Use group_by and summarize to count the number of trips for each unique route
route_counts <- non_round_trips %>% 
  group_by(City, Start.Station, End.Station) %>% 
  summarize(TripCount = n()) %>% 
  arrange(desc(TripCount))

# Select the top 5 routes for each city
top_routes <- route_counts %>% 
  group_by(City) %>% 
  top_n(5, TripCount)

Question 2¶

What percentage of trips in each city were taken by subscribers vs. customers?

In [9]:
# Use group_by and summarize to count the number of trips for each user type in each city, ignoring rows with missing User.Type values
user_counts <- bike_data %>% 
  filter(!is.na(User.Type)) %>%
  group_by(City, User.Type) %>% 
  summarize(TripCount = n()) %>% 
  ungroup()

# Use group_by and summarize to count the total number of trips in each city
total_trips <- bike_data %>% 
  group_by(City) %>% 
  summarize(TotalTrips = n()) %>% 
  ungroup()

# Merge the user counts and total trips data frames
user_percentages <- left_join(user_counts, total_trips, by = "City") %>% 
  mutate(Percentage = round(TripCount / TotalTrips * 100, 2))

Question 3¶

Does the amount of bike stations for each city correlate to a higher average number of trips?

In [12]:
# Create a data frame with the number of trips and stations in each city
city_summary <- bike_data %>%
  group_by(City) %>%
  summarize(TotalTrips = n(), TotalStations = n_distinct(Start.Station)) %>%
  ungroup() %>%
  mutate(TripsPerStation = TotalTrips / TotalStations)

# Generate a pie chart of the average trips per station for each city
ggplot(city_summary, aes(x = "", y = TripsPerStation, fill = City)) +
  geom_bar(stat = "identity", width = 1) +
  coord_polar("y", start = 0) +
  labs(
    title = "Average Trips per Station in Each City",
    x = NULL,
    y = "Average Trips per Station",
    fill = "City"
  ) +
  scale_fill_manual(values = cb_pal) +
  theme_void()
